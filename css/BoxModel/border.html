# 边框

# 边框规则属性

# 颜色

# 位置

/* 上 */
border-top-color ：<color>
/* 右 */
border-right-color ：<color>
/* 下 */
border-bottom-color ：<color>
/* 左 */
border-left-color ：<color>
属性值 描述
<color> 颜色值 数据类型

# 简写属性

border-color：<color>{1,4}

/* <color> 值 */
border-color: red;

/* 水平 | 垂直 */
border-color: red #f015ca;

/* 上 | 垂直 | 下 */
border-color: red rgb(240,30,50,.7) green;

/* 上 | 右 | 下 | 左 */
border-color: red yellow green blue;
属性值 描述
<color> 颜色值 数据类型 边框颜色

# 宽度

# 位置

/* 上 */
border-top-width ：<length> | thin | medium | thick
/* 右 */
border-right-width ：<length> | thin | medium | thick
/* 下 */
border-bottom-width ：<length> | thin | medium | thick
/* 左 */
border-left-width ：<length> | thin | medium | thick
属性值 描述
<length> 长度值 数据类型 设置宽度
thin 细线
medium 中线
thick 粗线

# 简写属性

border-width：<line-width>{1,4}
where
<line-width> = <length> | thin | medium | thick

/* 单值 */
border-width: 5px;
/* 水平 | 垂直 */
border-width: 2px 1.5em;
/* 上 | 垂直 | 下 */
border-width: 1px 2em 1.5cm;
/* 上 | 右 | 下 | 左 */
border-width: 1px 2em 0 4rem;
属性值 描述
<length> 长度值 数据类型 设置宽度
thin 细线
medium 中线
thick 粗线

# 样式

# 位置

/* 上 */
border-top-style ：<line-style>
/* 右 */
border-right-style ：<line-style>
/* 下 */
border-bottom-style ：<line-style>
/* 左 */
border-left-style ：<line-style>
属性值 描述
<line-style> 边框样式

# 简写属性

border-style：<line-style>{1,4}
where
<line-style> = none | hidden | dotted | dashed | solid | double | groove | ridge | inset | outset

/* 单值 */
border-style: hidden;
/* 水平 | 垂直 */
border-style: dotted solid;
/* 上 | 垂直 | 下 */
border-style: hidden double dashed;
/* 上 | 右 | 下 | 左 */
border-style: none solid dotted dashed;
属性值 描述
none 像 hidden 关键字一样，不显示边框
hidden 像 none 关键字一样，不显示边框，有隐藏效果
dotted 圆点
dashed 虚线
solid 实线
double 双实线
groove 雕刻效果的边框，样式与 ridge 相反
ridge 浮雕效果的边框，样式与 groove 相反
inset 陷入效果的边框，样式与 outset 相反
outset 突出效果的边框，样式与 inset 相反

# 边框方向

# 方向位置

/* 上 */
border-top ：<line-width> || <line-style> || <color>
/* 右 */
border-right ：<line-width> || <line-style> || <color>
/* 下 */
border-bottom ：<line-width> || <line-style> || <color>
/* 左 */
border-left ：<line-width> || <line-style> || <color>
属性值 描述
<line-width> 边框宽度
<line-style> 边框样式
<color> 边框颜色

# 简写属性

border：<line-width> || <line-style> || <color>

/* 样式 */
border: solid;

/* 宽度 | 样式 */
border: 2px dotted;

/* 样式 | 颜色 */
border: outset #f33;

/* 宽度 | 样式 | 颜色 */
border: medium dashed green;
属性值 描述
<line-width> 边框宽度
<line-style> 边框样式
<color> 边框颜色

# 边框圆角

# 左下角圆角

border-bottom-left-radius ： <length-percentage>{1,2}
where
<length-percentage> = <length> | <percentage>

/* 角边界的圆的半径*/
/* 单值 */
border-bottom-left-radius: 3px;
border-bottom-left-radius: 20%;

/*双值 水平方向和垂直方向 */
border-bottom-left-radius: 20% 20%;
border-bottom-left-radius: 20% 10%;
border-bottom-left-radius: 0.5em 1em;
属性值 描述
<length> 长度值数据类型 圆角
<percentage> 百分比数据类型 圆角

# 右下角圆角

border-bottom-right-radius ： <length-percentage>{1,2}
where
<length-percentage> = <length> | <percentage>

/* 角边界的圆的半径*/
/* 单值 */
border-bottom-right-radius: 3px;
border-bottom-right-radius: 20%;

/*双值 水平方向和垂直方向 */
border-bottom-right-radius: 20% 20%;
border-bottom-right-radius: 20% 10%;
border-bottom-right-radius: 0.5em 1em;
属性值 描述
<length> 长度值数据类型 圆角
<percentage> 百分比数据类型 圆角

# 左上角圆角

border-top-left-radius ： <length-percentage>{1,2}
where
<length-percentage> = <length> | <percentage>

/* 角边界的圆的半径*/
/* 单值 */
border-top-left-radius: 3px;
border-top-left-radius: 20%;

/*双值 水平方向和垂直方向 */
border-top-left-radius: 20% 20%;
border-top-left-radius: 20% 10%;
border-top-left-radius: 0.5em 1em;
属性值 描述
<length> 长度值数据类型 圆角
<percentage> 百分比数据类型 圆角

# 右上角圆角

border-top-right-radius ： <length-percentage>{1,2}
where
<length-percentage> = <length> | <percentage>

/* 角边界的圆的半径*/
/* 单值 */
border-top-right-radius: 3px;
border-top-right-radius: 20%;

/*双值 水平方向和垂直方向 */
border-top-right-radius: 20% 20%;
border-top-right-radius: 20% 10%;
border-top-right-radius: 0.5em 1em;
属性值 描述
<length> 长度值数据类型 圆角
<percentage> 百分比数据类型 圆角

# 简写属性

border-radius ： <length-percentage>{1,4} [ / <length-percentage>{1,4} ]?
where
<length-percentage> = <length> | <percentage>

/* 单值 */
/* 4个角设置半径 */
border-radius: 10px;

/* top-left和bottom-right | top-right和bottom-left */
border-radius: 10px 5%;

/* top-left | top-right 和 bottom-left | bottom-right */
border-radius: 2px 4px 2px;

/* top-left | top-right | bottom-right | bottom-left */
border-radius: 1px 0 3px 4px;

/* 水平/垂直 */
border-radius: 10px / 20px;

/* 水平 top-left和bottom-right | top-right和bottom-left
   /垂直 top-left-and-bottom-right | top-right-and-bottom-left */
border-radius: 10px 5% / 20px 30px;

/* 水平 top-left | top-right 和 bottom-left | bottom-righ /
   垂直 top-left | top-right-and-bottom-left | bottom-right */
border-radius: 10px 5px 2em / 20px 25px 30%;

/* 水平 top-left和bottom-right | top-right和bottom-left  /
   垂直 top-left | top-right | bottom-right | bottom-left */
border-radius: 10px 5% / 20px 25em 30px 35em;
属性值 描述
<length> 长度值数据类型 圆角
<percentage> 百分比数据类型 圆角

# 边框图像

# 边框图像超出

边框图像可超出边框盒的大小

border-image-outset ： [ <length> | <number> ]{1,4}

/* 单值 四面超出 */
border-image-outset: 1rem;
border-image-outset: 1.5;

/* 垂直 | 水平 */
border-image-outset: 1 1.2;

/* 上 | 左右 | 下 */
border-image-outset: 30px 2 45px;

/* 上 | 右 | 下 | 左 */
border-image-outset: 7px 12px 14px 5px;
属性值 描述
<length> 长度值数据类型 超出长度值
<percentage> 百分比数据类型 超出百分比

# 边框图像平铺

border-image-repeat ：[ stretch | repeat | round | space ]{1,2}

/* 单值 */
border-image-repeat: stretch;
border-image-repeat: repeat;
border-image-repeat: round;
border-image-repeat: space;

/* 垂直 | 水平 */
border-image-repeat: round stretch;
属性值 描述
stretch 图像拉伸撑满边框
repeat 图像平铺
round 图像等比例之间以拉伸进行图像平铺
space 图像等比例之间以间距进行图像平铺

# 边框图像切片

border-image-slice ： <number-percentage>{1,4} && fill?
where
<number-percentage> = <number> | <percentage>

/* 全切片 */
border-image-slice: 30%;

/* 垂直 | 水平 */
border-image-slice: 10% 30%;

/* 上 | 水平 | 下 */
border-image-slice: 30 30% 45;

/* 上 | 右 | 下 | 左 */
border-image-slice: 7 12 14 5;

/* 使用`fill`关键字*/
border-image-slice: 10% fill 7 12;
属性值 描述
<number> 长度值数据类型 切片图像偏移像素比
<percentage> 百分比数据类型 切片图像百分比
fill 保留中间图像区域并将其显示为背景图像

# 边框图像来源

border-image-source ：none | <image>
属性值 描述
none 不使用边框图像。border-style改为显示定义的外观
<image> 边框的图像 数据类型

# 边框图像宽度

border-image-width ：[ <length-percentage> | <number> | auto ]{1,4}
where
<length-percentage> = <length> | <percentage>

/* 关键字 */
border-image-width: auto;

/* 长度 */
border-image-width: 1rem;

/* 百分比 */
border-image-width: 25%;

/* 数值 */
border-image-width: 3;

/* 垂直 | 水平 */
border-image-width: 2em 3em;

/* 上 | 横向 | 下 */
border-image-width: 5% 15% 10%;

/* 上 | 右 | 下 | 左 */
border-image-width: 5% 2em 10% auto;
属性值 描述
<length> 长度值数据类型 设置宽度
<percentage> 百分比数据类型 设置宽度
<number> 数字数据类型 设置宽度倍数
auto border-image-slice相应尺寸大小

# 简写属性

border-image: <'border-image-source'> || <'border-image-slice'> [ / <'border-image-width'> | / <'border-image-width'>? / <'border-image-outset'> ]? || <'border-image-repeat'>

/* source | slice */
border-image: linear-gradient(red, blue) 27;

/* source | slice | repeat */
border-image: url("/images/border.png") 27 space;

/* source | slice | width */
border-image: linear-gradient(red, blue) 27 / 35px;

/* source | slice | width | outset | repeat */
border-image: url("/images/border.png") 27 23 / 50px 30px / 1rem round space
属性值 描述
<border-image-source> 图像来源
<border-image-slice> 图像切片 （上下左右）
<border-image-width> 图像宽度（上下左右）
<border-image-outset> 边框图像与元素外部边缘的距离（上下左右）
<border-image-repeat> 边框图像平铺
