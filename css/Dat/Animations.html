# 动画

# 规则

# 规定动画名称

@keyframes <keyframes-name> {
  <keyframe-block-list>
}
where 
<keyframes-name> = <custom-ident> | <string>
<keyframe-block-list> = <keyframe-block>+

where 
<keyframe-block> = <keyframe-selector># {
  <declaration-list>
}
where 
<keyframe-selector> = from | to | <percentage>


@keyframes  动画帧名称{
动画关键帧列表
}

/*  from  to 关键字动画 */
@keyframes slidein {
  from {
    transform: translateX(0%); 
  }

  to {
    transform: translateX(100%);
  }
}
/*  属性个数不定 */
@keyframes identifier {
  0% { top: 0; left: 0; }
  30% { top: 50px; }
  68%, 72% { left: 50px; }
  100% { top: 100px; left: 100%; }
}

/* 同一关键帧中的相同属性被重复定义*/
@keyframes identifier {
  0% { top: 0; }
  50% { top: 30px; left: 20px; }
  50% { top: 10px; }
  100% { top: 0; }
}

/* 关键帧中的 !important */
@keyframes important1 {
  from { margin-top: 50px; }
  50%  { margin-top: 150px !important; } /* 忽略 */
  to   { margin-top: 100px; }
}

@keyframes important2 {
  from { margin-top: 50px;
         margin-bottom: 100px; }
  to   { margin-top: 150px !important; /* 忽略 */
         margin-bottom: 50px; }
}
属性值 描述
<custom-ident> 字符串 数据类型 动画帧名称
from 起始0%
to 结束100%
<percentage> 百分比数据类型 动画序列中触发关键帧的时间点

# 动画属性

# 动画延迟时间

动画于何时开始，即从动画应用在元素上到动画开始的这段时间的长度。

animation-delay: <time>#

animation-delay: 3s;
animation-delay: 2s, 4ms;
属性值 描述
<time> 时间 数据类型 动画开始前等待的时间

# 动画播放方向

animation-direction :  <single-animation-direction>#
where 
<single-animation-direction> = normal | reverse | alternate | alternate-reverse
属性值 描述
normal 正常播放
reverse 反向播放
alternate 奇数循环次数正向播放，在偶数循环次数反向播放
alternate-reverse 奇数循环次数反向播放，在偶数循环次数正向播放

# 动画持续时间

animation-duration : <time>#

animation-duration: 6s
animation-duration: 1s, 15s
属性值 描述
<time> 时间 数据类型 动画持续时间的生命周期

# 动画填充模式

动画在执行之前和之后如何将样式应用于其目标

animation-fill-mode : <single-animation-fill-mode>#
where 
<single-animation-fill-mode> = none | forwards | backwards | both
属性值 描述
normal 动画执行之前和之后不会应用任何样式到目标元素
forwards 目标将保留由执行期间遇到的最后一个关键帧计算值。
最后一个关键帧取决于animation-direction和animation-iteration-count的值
backwards 动画将在应用于目标时立即应用第一个关键帧中定义的值，
并在animation-delay期间保留此值。 第一个关键帧取决于animation-direction的值
both 动画将遵循forwards和backwards的规则，从而在两个方向上扩展动画属性

forwards的属性

动画播放方向 动画迭代次数 最后一个关键帧
normal even or odd 100% or to
reverse even or odd 0% or from
alternate even 0% or from
alternate odd 100% or to
alternate-reverse even 100% or to
alternate-reverse odd 0% or from

backwards的属性

动画播放方向 第一个关键帧
normal or alternate 0% or from
reverse or alternate-reverse 100% or to

# 动画迭代次数

animation-iteration-count: <single-animation-iteration-count>#
where 
<single-animation-iteration-count> = infinite | <number>
属性值 描述
infinite 无限次播放
<number> 数字数据类型 默认值为1，播放次数

# 动画规定名称

应用的一系列动画，每个名称代表一个由@keyframes定义的动画序列

animation-iteration-count: [ none | <keyframes-name> ]#
where 
<keyframes-name> = <custom-ident> | <string>
属性值 描述
none 无关键帧名称
<custom-ident> 字符串数据类型 关键帧名称
<string> 字符数据类型 关键帧名称

# 动画运行状态

animation-play-state: <single-animation-play-state>#
where 
<single-animation-play-state> = running | paused
属性值 描述
running 当前动画正在运行
paused 当前动画已被停止

# 动画运行曲线速度

animation-timing-function: <timing-function>#
属性值 描述
<timing-function> 定时曲线数据类型

# 简写属性

animation ： <single-animation>#
where 
<single-animation> = <time> || <timing-function> || <time> || <single-animation-iteration-count> || <single-animation-direction> || <single-animation-fill-mode> || <single-animation-play-state> || [ none | <keyframes-name> ]

where 
<timing-function> = linear | <cubic-bezier-timing-function> | <step-timing-function>
<single-animation-iteration-count> = infinite | <number>
<single-animation-direction> = normal | reverse | alternate | alternate-reverse
<single-animation-fill-mode> = none | forwards | backwards | both
<single-animation-play-state> = running | paused
<keyframes-name> = <custom-ident> | <string>

where 
<cubic-bezier-timing-function> = ease | ease-in | ease-out | ease-in-out | cubic-bezier(<number <a href="/zh-CN/docs/CSS/Value_definition_syntax#Brackets" title="Brackets: enclose several entities, combinators, and multipliers to transform them as a single component">[0,1]>, <number>, <number <a href="/zh-CN/docs/CSS/Value_definition_syntax#Brackets" title="Brackets: enclose several entities, combinators, and multipliers to transform them as a single component">[0,1]>, <number>)
<step-timing-function> = step-start | step-end | steps(<integer>[, <step-position>]?)

where 
<step-position> = jump-start | jump-end | jump-none | jump-both | start | end



/* @keyframes 持续时间 | 定时曲线速度 | 延迟时间 | 
迭代次数 | 播放方向 | 填充模式 | 运行状态 | 关键帧名称 */
animation: 3s ease-in 1s 2 reverse both paused slidein;

/* @keyframes 关键帧名称 | 持续时间 | 定时曲线速度 | 延迟时间 */
animation: slidein 3s linear 1s;

/* @keyframes 关键帧名称 | 持续时间 */
animation: slidein 3s;
属性值 默认值 描述
animation-name none 关键帧名称
animation-duration 0s 持续时间
animation-timing-function ease 定时曲线速度
animation-delay 0s 延迟时间
animation-iteration-count 1 迭代次数
animation-direction normal 播放方向
animation-fill-mode none 填充模式
animation-play-state running 运行状态
